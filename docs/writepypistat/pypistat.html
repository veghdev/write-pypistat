147        self,
148        stat_type: Union[PypiStatType, str],
149        start_date: Optional[str] = None,
150        end_date: Optional[str] = None,
151    ) -> Optional[pd.DataFrame]:
152        """
153        Returns pypi statistics.
154
155        Args:
156            stat_type: Type of the statistics.
157            start_date (optional): Start date of the statistics,
158                should be in one of the following formats:
159                - %Y, for example 2022 will be 2022-01-01.
160                - %Y-%m, for example 2022-01 will be 2022-01-01.
161                - %Y-%m-%d, for example 2022-01-01 will be 2022-01-01.
162                - None will be the earliest available date.
163            end_date (optional): End date of the statistics,
164                should be in one of the following formats:
165                - %Y, for example 2022 will be 2022-12-31.
166                - %Y-%m, for example 2022-12 will be 2022-12-31.
167                - %Y-%m-%d, for example 2022-12-31 will be 2022-12-31.
168                - None will be the actual date.
169
170        Returns:
171            The gathered pypi statistics.
172        """
173
174        stat_date = StatDate(start=start_date, end=end_date)
175        return self._get_pypistat(PypiStatType(stat_type).value, stat_date)

177    def _get_pypistat(
178        self, stat_type: str, stat_date: StatDate
179    ) -> Optional[pd.DataFrame]:
180        method = getattr(pypistats, stat_type)
181        stats = None
182        try:
183            stats = method(
184                self._package_name,
185                total=True,
186                start_date=stat_date.start.strftime("%Y-%m-%d"),
187                end_date=stat_date.end.strftime("%Y-%m-%d"),
188                format="pandas",
189            )
190        except ValueError:
191            return stats
192        except IndexError:
193            return stats
194        if self.drop_percent_column:
195            stats.drop(["percent"], inplace=True, axis=1, errors="ignore")
196        stats.sort_values(["date", "category"], inplace=True)
197        if self.drop_total_row and stats.iloc[-1]["date"] is None:
198            stats = stats.head(-1)
199        if self.write_package_name:
200            stats.insert(0, "package", self._package_name)
201        date_index = stats.pop("date")
202        stats.insert(0, date_index.name, date_index)
203        stats = WritePypiStat._set_data_frame_columns(stats)
204        return stats

206    def _get_pypistat_by_none(
207        self, stat_type: str, stat_date: StatDate, postfix: Optional[str] = None
208    ) -> List[Dict[str, Union[str, pd.DataFrame, None]]]:
209        stats = []
210        stat_file = postfix if postfix is not None else "pypistat" + "_" + stat_type
211        stat_file += ".csv"
212        stat = self._get_pypistat(stat_type, stat_date)
213        if self.merge_stored_data:
214            stat = WritePypiStat._concat_with_stored_pypistat(
215                stat,
216                self._get_stored_pypistat(stat_file),
217                ["date", "package", "category"]
218                if self.write_package_name
219                else ["date", "category"],
220            )
221        if self.fill_no_data:
222            stat = self._concat_with_no_data(stat, stat_date)
223        stats.append({"stat": stat, "stat_file": stat_file})
224        return stats

226    def _get_pypistat_by_year(
227        self,
228        stat_type: str,
229        start_date: datetime,
230        end_date: datetime,
231        postfix: Optional[str] = None,
232    ) -> List[Dict[str, Union[str, pd.DataFrame, None]]]:
233        stats = []
234        years = []
235        actual_start_date = start_date
236        actual_year_end = None
237        actual_end_date = None
238        for i in range((end_date - start_date).days + 1):
239            day = start_date + timedelta(days=i)
240            stat_file = day.strftime("%Y") + "_"
241            stat_file += (
242                postfix if postfix is not None else "pypistat" + "_" + stat_type
243            )
244            stat_file += ".csv"
245            if day.year not in years:
246                years.append(day.year)
247                actual_year_end = datetime(day.year, 12, 31)
248                actual_end_date = (
249                    actual_year_end if actual_year_end <= end_date else end_date
250                )
251                stat_date = StatDate(
252                    start=actual_start_date.strftime("%Y-%m-%d"),
253                    end=actual_end_date.strftime("%Y-%m-%d"),
254                )
255                stat = self._get_pypistat(stat_type, stat_date)
256                if self.merge_stored_data:
257                    stat = WritePypiStat._concat_with_stored_pypistat(
258                        stat,
259                        self._get_stored_pypistat(stat_file),
260                        ["date", "package", "category"]
261                        if self.write_package_name
262                        else ["date", "category"],
263                    )
264                if self.fill_no_data:
265                    stat = self._concat_with_no_data(stat, stat_date)
266                stats.append({"stat": stat, "stat_file": stat_file})
267                actual_start_date = actual_year_end + timedelta(days=1)
268        return stats

358    def _get_stored_pypistat(self, stat_file: str) -> Optional[pd.DataFrame]:
359        stat = None
360        if self.outdir:
361            stat_file_path = os.path.join(self.outdir, stat_file)
362            if os.path.exists(stat_file_path):
363                stat = pd.read_csv(
364                    stat_file_path, dtype={"downloads": int, "category": str}

366                stat = WritePypiStat._set_data_frame_columns(stat)

393    def _concat_with_no_data(
394        self, stat: Optional[pd.DataFrame], stat_date: StatDate
395    ) -> pd.DataFrame:
396        days = WritePypiStat._get_days(stat_date)
397
398        no_data = {
399            "date": days,
400            "package": self._package_name,
401            "category": np.nan,
402            "downloads": 0,
403        }
404
405        if not self.write_package_name:
406            del no_data["package"]
407
408        if stat is not None:
409            del no_data["category"]
410            no_data_df = pd.DataFrame(data=no_data)
411            no_data_df = WritePypiStat._merge_data_frames(
412                stat,
413                no_data_df,
414                ["date", "package"] if self.write_package_name else ["date"],
415            )
416        else:
417            no_data_df = pd.DataFrame(data=no_data)
418            no_data_df = WritePypiStat._set_data_frame_columns(no_data_df)
419
420        return no_data_df

422    @staticmethod
423    def _merge_data_frames(
424        dataf1: pd.DataFrame, dataf2: pd.DataFrame, keys: List[str]
425    ) -> pd.DataFrame:
426        dataf1 = dataf1.replace("null", np.nan)
427        dataf1 = dataf1.replace("nan", np.nan)
428        dataf1 = WritePypiStat._set_data_frame_columns(dataf1)
429        dataf2 = dataf2.replace("null", np.nan)
430        dataf2 = dataf2.replace("nan", np.nan)
431        dataf2 = WritePypiStat._set_data_frame_columns(dataf2)
432        merged = pd.merge(dataf1, dataf2, how="outer", on=[*keys])
433        merged.sort_values([*keys], inplace=True)
434        if "downloads_y" in merged.columns:
435            merged.rename(columns={"downloads_x": "downloads"}, inplace=True)
436            merged.rename(columns={"downloads_y": "downloads_x"}, inplace=True)
437        merged.downloads.fillna(merged.downloads_x, inplace=True)
438        merged.drop(["downloads_x"], inplace=True, axis=1, errors="ignore")
439        return merged

441    @staticmethod
442    def _set_data_frame_columns(dataf: pd.DataFrame) -> pd.DataFrame:
443        for col in dataf.columns:
444            if col != "downloads":
445                dataf[col] = dataf[col].astype(str)
446            else:
447                dataf[col] = dataf[col].astype(int)
448        return dataf

450    @staticmethod
451    def _filter_data_frame_rows(stat: pd.DataFrame) -> pd.DataFrame:
452        stat = WritePypiStat._set_data_frame_columns(stat)
453        stat = stat.drop_duplicates()
454        df_zero = stat[stat["downloads"] == 0]
455        df_no_zero = stat[stat["downloads"] != 0]
456        stat = stat.drop(df_zero.index[df_zero["date"].isin(df_no_zero["date"])])
457        return stat

462        start_date: Optional[str] = None,

467        Writes pypi statistics into csv files.

489    def _write_pypistats(
490        self,
491        stat_type: str,
492        stat_date: StatDate,
493        postfix: Optional[str],
494    ) -> None:
495        stats = []
496        if self.date_period == StatPeriod.DAY:
497            stats += self._get_pypistat_by_day(
498                stat_type, stat_date.start, stat_date.end, postfix
499            )
500        elif self.date_period == StatPeriod.MONTH:
501            stats += self._get_pypistat_by_month(
502                stat_type, stat_date.start, stat_date.end, postfix
503            )
504        elif self.date_period == StatPeriod.YEAR:
505            stats += self._get_pypistat_by_year(
506                stat_type, stat_date.start, stat_date.end, postfix
507            )
508        else:
509            stats += self._get_pypistat_by_none(stat_type, stat_date, postfix)
510
511        for stat in stats:
512            self._write_csv(stat["stat"], stat["stat_file"])
513
514    def _write_csv(self, stat: Optional[pd.DataFrame], stat_file: str) -> None:
515        if stat is not None:
516            if self.outdir:
517                os.makedirs(self.outdir, exist_ok=True)
518                stat = WritePypiStat._filter_data_frame_rows(stat)
519                stat.to_csv(
520                    os.path.join(self.outdir, stat_file), index=False, encoding="utf-8"
521                )
